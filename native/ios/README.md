# Klip4ge — iOS Native Shell

This directory holds the iOS Xcode project generated by Capacitor.

## Setup (run once, requires macOS + Xcode 15+)

```bash
# From project root
npm install
npm run build

# Add iOS platform (generates this project)
npx cap add ios

# Sync web build into native shell
npx cap sync ios

# Open in Xcode
npx cap open ios
```

## Share Extension (iOS Share Sheet)
The Share Extension is a separate Xcode target that receives shares from other apps.

### Creating the Share Extension
1. In Xcode: File → New → Target → Share Extension
2. Name it `KlipShare`
3. Set Bundle ID: `app.klip4ge.app.KlipShare`
4. Add to same App Group: `group.app.klip4ge`

### ShareExtension/ShareViewController.swift (boilerplate)
```swift
import UIKit
import Social

class ShareViewController: SLComposeServiceViewController {
    override func isContentValid() -> Bool { return true }

    override func didSelectPost() {
        guard let extensionItems = extensionContext?.inputItems as? [NSExtensionItem] else {
            extensionContext?.completeRequest(returningItems: [], completionHandler: nil)
            return
        }

        for item in extensionItems {
            for provider in item.attachments ?? [] {
                if provider.hasItemConformingToTypeIdentifier("public.url") {
                    provider.loadItem(forTypeIdentifier: "public.url", options: nil) { url, _ in
                        if let url = url as? URL {
                            // Pass to main app via App Group UserDefaults
                            let defaults = UserDefaults(suiteName: "group.app.klip4ge")
                            defaults?.set(url.absoluteString, forKey: "pendingShareURL")
                            defaults?.set(self.contentText, forKey: "pendingShareTitle")
                            defaults?.set("ios_share", forKey: "pendingShareSource")
                            defaults?.synchronize()
                        }
                        DispatchQueue.main.async {
                            self.extensionContext?.completeRequest(returningItems: [], completionHandler: nil)
                        }
                    }
                }
            }
        }
    }

    override func configurationItems() -> [Any]! { return [] }
}
```

### AppDelegate.swift — Reading Share Extension Data
```swift
// In applicationDidBecomeActive or scene(_:openURLContexts:)
func checkSharedData() {
    let defaults = UserDefaults(suiteName: "group.app.klip4ge")
    if let url = defaults?.string(forKey: "pendingShareURL") {
        let title = defaults?.string(forKey: "pendingShareTitle") ?? ""
        let source = defaults?.string(forKey: "pendingShareSource") ?? "ios_share"
        
        // Clear immediately to prevent re-processing
        defaults?.removeObject(forKey: "pendingShareURL")
        defaults?.removeObject(forKey: "pendingShareTitle")
        defaults?.removeObject(forKey: "pendingShareSource")
        
        // Route to WebView via Capacitor bridge
        let deepLink = "klip4ge://share?url=\(url.addingPercentEncoding(withAllowedCharacters: .urlQueryAllowed) ?? "")&title=\(title.addingPercentEncoding(withAllowedCharacters: .urlQueryAllowed) ?? "")&source=\(source)"
        NotificationCenter.default.post(name: Notification.Name("KlipShareIntent"), object: deepLink)
    }
}
```
